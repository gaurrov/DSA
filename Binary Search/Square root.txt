class Solution {
    int floorSqrt(int n) {
        // code here
        if(n<2) return n;
        int left=1,right=n/2;
        int ans=0;
        while(left<=right){
            long mid=left+(right-left)/2;
            if(mid*mid<=n){
                ans=(int)mid;
                left=(int)mid+1;
            }else{
                right=(int)mid-1;
            }
        }
        return ans;
    }
}
