class Solution {
    public int nthRoot(int n, int m) {
        // code here
        if(m<=1) return m;
        int low=1,high=m;
        while(low<=high){
            int mid=low+(high-low)/2;
            long num=(long)Math.pow(mid,n);
            if(num==m) return mid;
            else if(num>m){
                high=mid-1;
            }else{
                low=mid+1;
            }
        }
        return -1;
    }
}
