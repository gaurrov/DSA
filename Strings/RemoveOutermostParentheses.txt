class Solution {
    public String removeOuterParentheses(String s) {
        int open=0;
        StringBuilder res = new StringBuilder();
        for(char ch:s.toCharArray()){
            if(ch=='(' && open++>0) {
                res.append('(');
            }
            if(ch==')' && open-->1){
                res.append(')');
            }
        }
        return res.toString();
    }
}
